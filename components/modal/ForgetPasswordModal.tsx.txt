"use client";

import { Modal, Stack, Text, TextInput, Button, useMantineTheme } from "@mantine/core";
import { IconMail, IconXboxXFilled } from "@tabler/icons-react";
import { useForm } from "@mantine/form";
import { useTranslations } from "next-intl";
import classes from "./ForgetPasswordModal.module.css";

import { validateEmail } from "../../../utils/validation";

interface ForgetPasswordModalProps {
  opened: boolean;
  onClose: (value?: any) => void;
  onSubmit: (email: string) => void;
}

export function ForgetPasswordModal({
  opened,
  onClose,
  onSubmit
}: ForgetPasswordModalProps) {
  const t = useTranslations('auth');
  const tValidation = useTranslations("validation");
  const theme = useMantineTheme();

  const form = useForm({
    initialValues: {
      email: "",
    },
    validate: {
      email: (value) => validateEmail(value, tValidation),
    },
  });

  const handleSubmit = () => {
    const validation = form.validate();
    if (!validation.hasErrors) {
      onSubmit(form.values.email); // Call parent function to handle API request
    }
  };

  return (
    <>
      <Modal
        opened={opened}
        onClose={onClose}
        title={<Text style={theme.headings.sizes.h2}>{t("forgotPass")}</Text>}
        overlayProps={{
          backgroundOpacity: 0.55,
          blur: 3,
        }}
        centered
      >
        <form onSubmit={(e) => {
          e.preventDefault();
          handleSubmit();
        }}>
          <Stack>
            <Text style={theme.other.textSmall}>
              {t("forgotPassDes")}
            </Text>
            <TextInput
              radius="xl"
              size="md"
              leftSection={
                <IconMail
                  size={20}
                  color={
                    form.errors.email
                      ? "var(--mantine-color-red-9)"
                      : "var(--mantine-color-blue-9)"
                  }
                />
              }
              rightSection={
                form.values.email ? (
                  <IconXboxXFilled
                    size={16}
                    style={{ cursor: "pointer" }}
                    onClick={() => form.setFieldValue("email", "")}
                  />
                ) : null
              }
              placeholder={t("placeHolderemail")}
              classNames={{
                input: classes.inputField,
                error: classes.inputError,
                wrapper: form.errors.email ? classes.inputErrorWrapper : undefined,
              }}
              {...form.getInputProps("email")}
              required
            />
            <Button
              radius="xl"
              size="md"
              className={classes.authButton}
              type="submit"
            >
              {t("resetPass")}
            </Button>
          </Stack>
        </form>
      </Modal>
    </>
  );
}
